75inline void url_search_params::set(const std::string_view key,
76 const std::string_view value) {
77 const auto find = [&key](auto &param) { return param.first == key; };
78
79 auto it = std::find_if(params.begin(), params.end(), find);
80
81 if (it == params.end()) {
82 params.emplace_back(key, value);
83 } else {
84 it->second = value;
85 params.erase(std::remove_if(std::next(it), params.end(), find),
86 params.end());
87 }
